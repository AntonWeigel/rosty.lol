    code: |-
      export function RenderSections({ sections }) {
        return sections?.map((s, i) => {
          switch (s._template) {
            case 'header': return <DocHeader key={i} title={s.title} />;
            case 'text': return <DocText key={i} content={s.content} />;
            case 'image': return <DocImage key={i} src={s.src} alt={s.alt} source={s.source} />;
            case 'infoBlock': return <DocCallout key={i} content={s.content} />;
            case 'codeBlock': return <DocCode key={i} language={s.language} code={s.code} />;
            default: return null;
          }
        });
      }
